  ```python
  import time, threading

  class NatsCircuitBreaker:
      def __init__(self, failure_threshold: int = 3, recovery_timeout: float = 60.0) -> None:
          self.failure_threshold = failure_threshold
          self.recovery_timeout = recovery_timeout
          self._failures = 0
          self._open_until: float = 0.0
          self._lock = threading.Lock()

      def is_open(self) -> bool:
          with self._lock:
              if self._open_until and time.monotonic() < self._open_until:
                  return True
              return False

      def record_success(self) -> None:
          with self._lock:
              self._failures = 0
              self._open_until = 0.0

      def record_failure(self) -> None:
          with self._lock:
              self._failures += 1
              if self._failures >= self.failure_threshold:
                  self._open_until = time.monotonic() + self.recovery_timeout
  ```
- **Verify:** `python -c "from lyra.nats.circuit_breaker import NatsCircuitBreaker; cb = NatsCircuitBreaker(failure_threshold=1); cb.record_failure(); assert cb.is_open()"`

  class NatsSttClient:
      def __init__(self, ...):
          ...
          self._cb = NatsCircuitBreaker()

      async def transcribe(self, path):
          if self._cb.is_open():
              raise STTUnavailableError("STT circuit open — adapter down")
          try:
              result = await self._do_transcribe(path)
              self._cb.record_success()
              return result
          except (STTUnavailableError, TimeoutError, Exception) as exc:
              self._cb.record_failure()
              raise
  ```

  async def probe_voice_services(nc: "NATS", stt: object | None, tts: object | None) -> None:
      from nats.aio.errors import NoRespondersError
      for name, subject in [
          ("STT", "lyra.voice.stt.request"),
          ("TTS", "lyra.voice.tts.request"),
      ]:
          client = stt if name == "STT" else tts
          if client is None:
              continue
          try:
              await nc.request(subject, b"", timeout=1.0)
          except (NoRespondersError, TimeoutError):
              log.warning("%s adapter not reachable at boot — will retry per-request", name)
          except Exception:
              pass
  ```
