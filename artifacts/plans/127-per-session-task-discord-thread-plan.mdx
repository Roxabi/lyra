## Summary

Replace `asyncio.Lock` in `Pool` with a per-session `asyncio.Task` (inbox queue + sequential process loop), add per-turn timeout and `/stop` cancellation, and implement Discord thread auto-creation on @mention in text channels. This is Slice 3 of #112 (PR C).

File: `src/lyra/core/pool.py`

Expected shape skeleton:
```python
from __future__ import annotations
import asyncio
import collections.abc
import logging
from collections import deque
from typing import TYPE_CHECKING

from anthropic import APIError as AnthropicAPIError

if TYPE_CHECKING:
    from lyra.core.hub import Hub
    from lyra.core.agent import AgentBase
from .message import Message, Response, GENERIC_ERROR_REPLY

log = logging.getLogger(__name__)

TURN_TIMEOUT_DEFAULT = 60.0


class Pool:
    """One pool per conversation scope. Holds history and a per-session asyncio.Task."""

    def __init__(
        self,
        pool_id: str,
        agent_name: str,
        hub: "Hub",
        turn_timeout: float = TURN_TIMEOUT_DEFAULT,
    ) -> None:
        self.pool_id = pool_id
        self.agent_name = agent_name
        self.history: list[Message] = []
        self.sdk_history: deque[dict] = deque()
        self.max_sdk_history: int = 50
        self._hub = hub
        self._turn_timeout = turn_timeout
        self._inbox: asyncio.Queue[Message] = asyncio.Queue()
        self._current_task: asyncio.Task | None = None

    def submit(self, msg: Message, agent: "AgentBase") -> None:
        """Enqueue msg; start processing task if not running."""
        self._inbox.put_nowait(msg)
        if self._current_task is None or self._current_task.done():
            self._current_task = asyncio.create_task(
                self._process_loop(), name=f"pool:{self.pool_id}"
            )

    def cancel(self) -> None:
        """Cancel the current processing task (no-op if idle)."""
        if self._current_task is not None and not self._current_task.done():
            self._current_task.cancel()

    async def _process_loop(self) -> None:
        """Consume inbox sequentially until empty."""
        while True:
            try:
                msg = self._inbox.get_nowait()
            except asyncio.QueueEmpty:
                break
            agent = self._hub.agent_registry.get(self.agent_name)
            if agent is None:
                log.error("no agent %r for pool %s — message dropped", self.agent_name, self.pool_id)
                continue
            try:
                await asyncio.wait_for(
                    self._process_one(msg, agent), timeout=self._turn_timeout
                )
            except asyncio.TimeoutError:
                _reply = self._hub._msg_manager.get("timeout") if self._hub._msg_manager else "Request timed out."
                await self._safe_dispatch(msg, Response(content=_reply))
            except asyncio.CancelledError:
                _reply = self._hub._msg_manager.get("cancelled") if self._hub._msg_manager else "Cancelled."
                await self._safe_dispatch(msg, Response(content=_reply))
                raise
            except Exception as exc:
                log.exception("unhandled error in pool %s: %s", self.pool_id, exc)
                _reply = self._hub._msg_manager.get("generic") if self._hub._msg_manager else GENERIC_ERROR_REPLY
                await self._safe_dispatch(msg, Response(content=_reply))
                if self._hub.circuit_registry is not None:
                    _cb = self._hub.circuit_registry.get("hub")
                    if _cb is not None:
                        _cb.record_failure()
        self._current_task = None

    async def _process_one(self, msg: Message, agent: "AgentBase") -> None:
        """Run agent.process and dispatch result. Records CB success."""
        result = agent.process(msg, self)
        if isinstance(result, collections.abc.AsyncIterator):
            try:
                await self._hub.dispatch_streaming(msg, result)
                self._record_cb_success()
            except BaseException as exc:
                self._record_cb_failure(exc)
                raise
        else:
            try:
                response = await result
                self._record_cb_success()
            except Exception as exc:
                self._record_cb_failure(exc)
                raise
            await self._hub.dispatch_response(msg, response)

    def _record_cb_success(self) -> None:
        if self._hub.circuit_registry is not None:
            for name in ("anthropic", "hub"):
                cb = self._hub.circuit_registry.get(name)
                if cb is not None:
                    cb.record_success()

    def _record_cb_failure(self, exc: BaseException) -> None:
        if self._hub.circuit_registry is not None:
            _hub_cb = self._hub.circuit_registry.get("hub")
            if _hub_cb is not None:
                _hub_cb.record_failure()
            if isinstance(exc, AnthropicAPIError):
                _ant_cb = self._hub.circuit_registry.get("anthropic")
                if _ant_cb is not None:
                    _ant_cb.record_failure()

    async def _safe_dispatch(self, msg: Message, response: Response) -> None:
        try:
            await asyncio.wait_for(self._hub.dispatch_response(msg, response), timeout=10)
        except Exception as exc:
            log.exception("_safe_dispatch failed for pool %s: %s", self.pool_id, exc)

    def extend_sdk_history(self, new_messages: list[dict]) -> None:
        self.sdk_history.extend(new_messages)
        while len(self.sdk_history) > self.max_sdk_history:
            self.sdk_history.popleft()
```

**T4 — Add /stop builtin to CommandRouter**
Phase: GREEN | Agent: backend-dev | Difficulty: 2 | Parallel-safe: Y
Spec trace: S4-5

File: `src/lyra/core/command_router.py`

Add to `_DEFAULT_BUILTINS`:
```python
"/stop": CommandConfig(builtin=True, description="Cancel the current processing turn"),
```

Add handler in `dispatch()` after `/circuit` check:
```python
if command_name == "/stop":
    if pool is not None:
        pool.cancel()
    return Response(content="")  # reply sent by Pool._process_loop on CancelledError
```

Note: the actual cancellation reply is sent by `Pool._process_loop` on `CancelledError`. The `/stop` dispatch just triggers the cancel and returns an empty response (which dispatch_response will skip or send as empty — confirm behavior). Actually, returning empty string will send an empty message. Better to not send a reply from dispatch since Pool will send the cancelled reply. We can return `Response(content="")` and skip dispatch if content is empty — but that requires hub changes.

**Alternative:** Return a sentinel or make CommandRouter aware. Simplest: just call `pool.cancel()` and return `Response(content="")` — the hub will attempt to send an empty message. This is a minor UX issue. Better: the hub can skip `dispatch_response` if `response.content` is empty.

**Decision:** Return `Response(content="")` and document that the cancelled reply comes from the Pool task. Hub already calls `dispatch_response` — empty content will send an empty message. To avoid this, add a check in `Hub.run()`:
```python
if response.content:
    await self.dispatch_response(msg, response)
```

This change goes in T2 (hub.py).
