    while True:

        key = self._state_key(event)
        new_state = self._to_state(event)
        prev = self._state_map.get(key)

Injection Pattern — Module-level Singleton
events.py — new file
# src/lyra/core/event_bus.py
_bus: EventBus | None = None

def get_event_bus() -> EventBus | None:
    return _bus

def set_event_bus(bus: EventBus) -> None:
    global _bus
    _bus = bus

# Usage in pool.py — no constructor churn
def emit(event: MonitoringEvent) -> None:
    bus = get_event_bus()
    if bus is not None:
        bus.emit(event)    # put_nowait — never blocks

    await self._run_loop()
finally:
    agg_task.cancel()           # clean shutdown
How to Add a New Event Emitter
3-step pattern — add any new emitter in < 5 lines
1
Import the event type and helper
from .event_bus import emit
from .events import AgentStarted   # or whichever event
2
Call emit() at the state transition point
async def _guarded_process_one(self, msg, agent):
    emit(AgentStarted(agent_id=self.pool_id))   # ← add here
    try:
        await asyncio.wait_for(
            self._process_one(msg, agent),
            timeout=self._turn_timeout
        )
3
No wiring needed — emit() finds the bus automatically
The module-level singleton means no constructor changes, no DI plumbing, no changes to call sites that don't emit events. Works for externally-created CircuitBreakers too.
Singleton vs Constructor Injection
Constructor Injection (rejected)
  • Every emitter (Pool, CircuitBreaker, InboundBus) needs event_bus parameter in __init__
  • CircuitBreaker is created externally by user code — hub can't pass bus at construction time
  • Tests must wire bus into every object under test
  • Hub.__init__ signature grows for each new feature
  • Refactor cost: 4+ files, all existing tests need updating
Module Singleton (chosen)
  • set_event_bus() called once in Hub.__init__ — done
  • CircuitBreaker imports emit as a free function — zero coupling to Hub
  • Tests: call set_event_bus(mock_bus) in fixture, reset in teardown
  • No constructor changes in Pool, CircuitBreaker, or InboundBus
  • Adding a new emitter is 2 import lines + 1 emit() call

EventBus — Non-blocking Guarantee
emit() is always synchronous + non-blocking
class EventBus:
    def emit(self, event: MonitoringEvent) -> None:
        """Fan-out to all subscriber queues. Never blocks. Never awaits."""
        for q in self._subscribers:
            try:
                q.put_nowait(event)      # raises QueueFull on overflow
            except asyncio.QueueFull:
                self.dropped_events += 1  # counter, never raises
The hot path in _guarded_process_one() is a tight asyncio loop. put_nowait is a single dict append — microseconds. Queue overflow drops silently and increments the counter (observable via event_bus.dropped_events).
Operator Experience: Before vs After
Before — External Cron Polling
  • Cron ticks every N seconds, calls /health HTTP endpoint
  • LLM invoked on every tick — even during hours of healthy operation
  • Noisy alerts: healthy system still triggers periodic "check" LLM calls
  • Token waste scales with uptime, not with problems
  • Race condition: cron sees stale HTTP snapshot, not live internal state
  • Cron is an external process — can die silently without affecting agents
  • No distinction between "agent failed" and "circuit open" in alert payload
After — Internal Event System
  • Events fired only on state transitions — zero noise during steady-state
  • LLM monitoring called only when something actually changes
  • EventAggregator dedup: repeated "AgentFailed" for same pool = one alert
  • Typed events carry rich context: agent_id, platform, failure_count, exc_type
  • No HTTP round-trip — events are in-process, sub-millisecond
  • Aggregator crash → falls back to cron gracefully (no silent failure)
  • Queue depth events are edge-triggered: one alert on crossing, one on recovery

Fallback Strategy
EventAggregator crash → transparent fallback to cron
# hub.py — aggregator task lifecycle
async def run(self) -> None:
    agg_task = asyncio.create_task(
        self._aggregator.run(), name="event-aggregator"
    )

    def _on_agg_done(t: asyncio.Task[None]) -> None:
        if not t.cancelled() and (exc := t.exception()):
            log.warning("event-aggregator exited: %s — falling back to cron", exc)
            # cron continues running; no auto-restart

    agg_task.add_done_callback(_on_agg_done)
    try:
        await self._run_loop()
    finally:
        agg_task.cancel()
        await asyncio.gather(agg_task, return_exceptions=True)

maxsize=1000 — Overflow Analysis
When can overflow happen?
  • ▸EventAggregator is blocked on its monitoring callback (LLM call)
  • ▸Very high message throughput + slow LLM response time
  • ▸Multiple concurrent pool failures emitting bursts of events
Behavior on overflow
  • ▸Event is silently dropped — hot path is never blocked
  • ▸event_bus.dropped_events counter increments
  • ▸Aggregator catches up when LLM call completes — backpressure absorbed
  • ▸Cron fallback sees the /health snapshot independently
