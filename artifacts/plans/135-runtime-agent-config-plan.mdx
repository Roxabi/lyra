**T-03** [GREEN] `_VALID_PARAMS` dict + validation helper
- **File:** `src/lyra/core/runtime_config.py`
- **Snippet:**
  ```python
  _VALID_PARAMS: dict[str, type] = {
      "style": str,
      "language": str,
      "temperature": float,
      "model": str,   # None sentinel: value "none" or "" resets to None
      "max_steps": int,
      "extra_instructions": str,
  }

  def set_param(rc: RuntimeConfig, key: str, value: str) -> RuntimeConfig:
      """Parse + validate a key=value pair, return new RuntimeConfig instance."""
      if key not in _VALID_PARAMS:
          raise ValueError(f"Unknown param: {key}. Valid: {', '.join(_VALID_PARAMS)}")
      if key == "style" and value not in _STYLES:
          raise ValueError(f"Invalid value for style: must be one of {sorted(_STYLES)}")
      if key == "temperature":
          try:
              fv = float(value)
          except ValueError:
              raise ValueError("Invalid value for temperature: must be float 0–1")
          if not 0.0 <= fv <= 1.0:
              raise ValueError("Invalid value for temperature: must be float 0–1")
          return replace(rc, temperature=fv)
      if key == "max_steps":
          try:
              iv = int(value)
          except ValueError:
              raise ValueError("Invalid value for max_steps: must be integer")
          return replace(rc, max_steps=iv)
      if key in ("model",) and value.lower() in ("none", ""):
          return replace(rc, model=None)
      return replace(rc, **{key: value})
  ```

  log = logging.getLogger(__name__)

  @classmethod
  def load(cls, path: Path) -> RuntimeConfig:
      if not path.exists():
          return cls()
      try:
          with path.open("rb") as f:
              data = tomllib.load(f)
      except Exception as exc:
          log.warning("Failed to load runtime config from %s: %s — using defaults", path, exc)
          return cls()
      rc = cls()
      for key, raw_value in data.items():
          try:
              rc = set_param(rc, key, str(raw_value))
          except ValueError as exc:
              log.warning("Ignoring invalid runtime config entry %r=%r: %s", key, raw_value, exc)
      return rc
  ```

  def _cmd_config_set(self, args: list[str]) -> Response:
      from lyra.core.runtime_config import set_param, _AGENTS_DIR
      updates: list[str] = []
      rc = self._runtime_config
      for token in args:
          if "=" not in token:
              return Response(content=f"Invalid format: {token!r}. Use key=value.")
          key, _, value = token.partition("=")
          try:
              rc = set_param(rc, key.strip(), value.strip())
              updates.append(f"{key} = {value}")
          except ValueError as exc:
              return Response(content=str(exc))
      runtime_file = _AGENTS_DIR / "lyra_runtime.toml"
      rc.save(runtime_file)
      self._runtime_config = rc      # replace, not mutate
      updated = ", ".join(updates)
      return Response(content=f"Updated: {updated}\nSaved to {runtime_file.name}")
  ```
