```python
"""Voice session management for DiscordAdapter (issue #255)."""
from __future__ import annotations

import enum
import queue
import shutil
from dataclasses import dataclass, field
from typing import TYPE_CHECKING

import discord

if TYPE_CHECKING:
    pass

_FRAME_SIZE = 3840  # 20 ms × 48 kHz × 2 ch × 2 bytes

class VoiceMode(enum.Enum):
    TRANSIENT = "transient"
    PERSISTENT = "persistent"

@dataclass
class VoiceSession:
    guild_id: str
    voice_client: discord.VoiceClient
    channel_id: int
    mode: VoiceMode
    source: "PCMQueueSource" = field(default_factory=lambda: PCMQueueSource())

    def is_active(self) -> bool:
        return self.voice_client.is_connected()

class PCMQueueSource(discord.AudioSource):
    def __init__(self) -> None:
        self._q: queue.Queue[bytes | None] = queue.Queue(maxsize=0)
        self._buf: bytes = b""

    def push(self, chunk: bytes) -> None:
        self._buf += chunk
        while len(self._buf) >= _FRAME_SIZE:
            self._q.put_nowait(self._buf[:_FRAME_SIZE])
            self._buf = self._buf[_FRAME_SIZE:]

    def push_eof(self) -> None:
        if self._buf:
            self._q.put_nowait(self._buf.ljust(_FRAME_SIZE, b"\x00"))
            self._buf = b""
        self._q.put_nowait(None)

    def read(self) -> bytes:
        try:
            item = self._q.get(timeout=0.1)
        except queue.Empty:
            return b"\x00" * _FRAME_SIZE  # silence frame while waiting
        if item is None:
            return b""
        return item

    def is_opus(self) -> bool:
        return False
```

```python
    def test_exact_frame(self) -> None:
        src = PCMQueueSource()
        src.push(b"x" * _FRAME)
        src.push_eof()
        assert src.read() == b"x" * _FRAME
        assert src.read() == b""

        src = PCMQueueSource()
        src.push(b"a" * (_FRAME * 2 + 100))
        src.push_eof()
        frames = []
        while True:
            f = src.read()
            if f == b"":
                break
            frames.append(f)
        assert len(frames) == 3
        assert all(len(f) == _FRAME for f in frames)
        assert frames[-1] == b"a" * 100 + b"\x00" * (_FRAME - 100)

    def test_silence_frame_on_empty_queue(self) -> None:
        src = PCMQueueSource()
        silence = src.read()
        assert silence == b"\x00" * _FRAME

    def test_eof_after_push_eof(self) -> None:
        src = PCMQueueSource()
        src.push_eof()
        assert src.read() == b""

    def test_is_opus_false(self) -> None:
```

```python
class VoiceDependencyError(Exception):
    """Raised when a required voice system dependency is missing."""

class VoiceAlreadyActiveError(Exception):
    """Raised when join() is called for a guild that already has an active session."""
    def __init__(self, guild_id: str) -> None:
        super().__init__(f"Voice session already active for guild {guild_id!r}")
        self.guild_id = guild_id

_deps_checked = False

def _check_voice_deps() -> None:
    """Verify libopus, ffmpeg, and discord.py[voice] are available. Lazy, runs once."""
    global _deps_checked
    if _deps_checked:
        return
    try:
        import discord.voice_client  # noqa: F401
    except ImportError as exc:
        raise VoiceDependencyError(
            "discord.py[voice] not installed — run: pip install discord.py[voice]"
        ) from exc
    try:
        discord.opus.load_opus()
    except Exception:
        if not discord.opus.is_loaded():
            raise VoiceDependencyError(
                "libopus not found — install libopus-dev (Ubuntu) or libopus (macOS)"
            )
    if shutil.which("ffmpeg") is None:
        raise VoiceDependencyError(
            "ffmpeg not found — install ffmpeg (apt install ffmpeg)"
        )
    _deps_checked = True
```

```python
import logging
log = logging.getLogger(__name__)

class VoiceSessionManager:
    """Manages per-guild Discord voice sessions."""

    def __init__(self, client: discord.Client) -> None:
        self._client = client
        self._sessions: dict[str, VoiceSession] = {}

    async def join(
        self,
        guild: discord.Guild,
        channel: discord.VoiceChannel,
        mode: VoiceMode,
    ) -> VoiceSession:
        guild_id = str(guild.id)
        if guild_id in self._sessions:
            raise VoiceAlreadyActiveError(guild_id)
        _check_voice_deps()
        source = PCMQueueSource()
        voice_client = await channel.connect()
        session = VoiceSession(
            guild_id=guild_id,
            voice_client=voice_client,
            channel_id=channel.id,
            mode=mode,
            source=source,
        )
        self._sessions[guild_id] = session
        log.info("Voice session joined: guild=%s channel=%s mode=%s", guild_id, channel.id, mode)
        return session

    async def leave(self, guild_id: str) -> None:
        session = self._sessions.get(guild_id)
        if session is None:
            return
        session.source.push_eof()
        await session.voice_client.disconnect()
        del self._sessions[guild_id]
        log.info("Voice session left: guild=%s", guild_id)

    def invalidate(self, guild_id: str) -> None:
        self._sessions.pop(guild_id, None)
        log.info("Voice session invalidated: guild=%s", guild_id)

    def get(self, guild_id: str) -> VoiceSession | None:
        return self._sessions.get(guild_id)
```

```python
async def on_voice_state_update(
    self,
    member: Any,
    before: discord.VoiceState,
    after: discord.VoiceState,
) -> None:
    """Invalidate stale voice session when the bot is forcibly disconnected."""
    bot_user = self._bot_user
    if bot_user is None or member.id != bot_user.id or after.channel is not None:
        return
    guild_id = str(member.guild.id)
    self._vsm.invalidate(guild_id)
```
