  @classmethod
  def from_adapter(
      cls,
      *,
      platform: Platform,
      bot_id: str,
      user_id: str,
      user_name: str,
      content: MessageContent,
      type: MessageType,
      timestamp: datetime,
      is_mention: bool = False,
      is_from_bot: bool = False,
      platform_context: TelegramContext | DiscordContext | None = None,
  ) -> "Message":
      """Construct a Message from an adapter. trust is always 'user'."""
      return cls(
          id=f"{platform.value}:{user_id}:{timestamp.timestamp():.0f}",
          platform=platform,
          bot_id=bot_id,
          channel=platform.value,  # deprecated alias
          user_id=user_id,
          user_name=user_name,
          content=content,
          type=type,
          timestamp=timestamp,
          is_mention=is_mention,
          is_from_bot=is_from_bot,
          platform_context=platform_context,
          trust="user",  # never caller-controlled
      )
  ```

  def _make_secret_verifier(webhook_secret: str):

  def _normalize(self, msg: aiogram.types.Message) -> Message:
      from aiogram.enums import MessageEntityType, ChatType
      ctx = TelegramContext(
          chat_id=msg.chat.id,
          topic_id=msg.message_thread_id,
          is_group=msg.chat.type != ChatType.PRIVATE,
      )
      is_mention = (
          msg.chat.type == ChatType.PRIVATE
          and False
          or any(
              e.type == MessageEntityType.MENTION
              and (msg.text or "")[e.offset:e.offset + e.length].lstrip("@") == self._bot_username
              for e in (msg.entities or [])
          )
      )
      return Message.from_adapter(
          platform=Platform.TELEGRAM,
          bot_id=self._bot_id,
          user_id=f"tg:user:{msg.from_user.id}" if msg.from_user else "tg:user:unknown",
          user_name=msg.from_user.full_name if msg.from_user else "",
          content=TextContent(text=msg.text or ""),
          type=MessageType.TEXT,
          timestamp=msg.date,
          is_mention=is_mention,
          is_from_bot=bool(msg.from_user and msg.from_user.is_bot),
          platform_context=ctx,
      )
  ```

      if msg.from_user and msg.from_user.is_bot:

      hub_msg = self._normalize(msg)
      if self._hub.bus.full():
